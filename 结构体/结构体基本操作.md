# 结构体基本操作

### 1.结构体的基本操作
```
#include "stdio.h"
#include "stdlib.h"
#include "string.h"

//结构体变量四字节对齐
typedef struct _teacher
{
    char name[62];
    int age;
}teacher;
void main()
{
    struct _teacher *p = NULL;
    //结构体变量的初始化
    teacher t1 = {"ddd",55};
    p = &t1;
    //得到结构体变量的值得两种方法（指针和变量.）
    printf("name:%s,age:%d\n",p->name,t1.age);
    printf("%d\n",sizeof(t1));
    system("pause");
}
```

### 3.结构体数组
```
#include "stdio.h"
#include "stdlib.h"
#include "string.h"
typedef struct _teacher
{
    char name[62];
    int age;
}teacher;

void PrintfArray(teacher *p, int num)
{
    int i = 0;
    for(i=0; i<num; i++)
    {
        printf("%d\n",p[i].age);
    }
}
int SortfArray(teacher *p, int num)
{
    teacher tmp;//注意这里不是teacher *tmp,而是应该这样
    int i=0, j = 0;
    for(i=0; i<num; i++)
    {
        for(j=0; j<num;j++)
        {
            if(p[i].age < p[j].age)
            {
                tmp = p[i];
                p[i] = p[j];
                p[j] = tmp;
            }
        }
    }
}
void main()
{
    int i = 0;
    teacher tArray[10];
    for(i=0; i<4; i++)
    {
        printf("请输入年龄：");
        scanf("%d",&tArray[i].age);
    }
    PrintfArray(tArray , 4);
    SortfArray(tArray ,4);
    printf("排序之后：\n");
    PrintfArray(tArray , 4);
}
```
